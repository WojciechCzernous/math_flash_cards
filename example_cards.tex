\section{Quadratic Equations}
%========================================

\begin{flashcard}{quad-001}
\Q{Solve: $x^2 - 5x + 6 = 0$}
\A{$x = 2$ or $x = 3$}
\end{flashcard}
%@rep:quad-001:2026-01-30:1:2.5

\begin{flashcard}{quad-002}
\Q{What is the quadratic formula?}
\A{$x = \displaystyle\frac{-b \pm \sqrt{b^2 - 4ac}}{2a}$}
\end{flashcard}

\begin{flashcard}{quad-003}
\Q{Factor: $x^2 - 9$}
\A{$(x+3)(x-3)$}
\end{flashcard}

%========================================
\section{Polynomials}
%========================================

\begin{flashcard}{poly-001}
\Q{Simplify: $\displaystyle\frac{x^2 - 4}{x - 2}$}
\A{$x + 2$ \quad (where $x \neq 2$)}
\end{flashcard}

\begin{flashcard}{poly-002}
\Q{Expand: $(a + b)^3$}
\A{$a^3 + 3a^2b + 3ab^2 + b^3$}
\end{flashcard}

%========================================
\section{Exponents and Logarithms}
%========================================

\begin{flashcard}{exp-001}
\Q{$\log_a(xy) = \;?$}
\A{$\log_a x + \log_a y$}
\end{flashcard}

\begin{flashcard}{exp-002}
\Q{$a^m \cdot a^n = \;?$}
\A{$a^{m+n}$}
\end{flashcard}
